int password()
{

   {

   }

  lcd.print("Enter Password:");
  delay(2000);

  int pass_word[7];
  int index=0;
  while(index<7)
  {
     custom_key = keypad.getKey();
    if(custom_key)
    {
      lcd.setCursor(index,1);
      pass_word[index]=custom_key;
      lcd.print("*");
      index++;
      Serial.print(custom_key);
    }
  }
    lcd.clear();
    int check=1;
    for(int i=0;i<=5;i++)
    {
      if(pass_word[i]-48==Master[i]){
        check*1;}
      
       else{
      check=check*0;

      Serial.print(" ");
      Serial.print(Master[i]);
      Serial.print(" ");
      Serial.println("Wrong");}
    }

    if(check == 1)
    {
      lcd.print("Correct");
      delay(2000);

      clearData();

 }
